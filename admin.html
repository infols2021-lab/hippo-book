Политики безопасности (RLS) — инструкция

Ниже — SQL, который нужно применить в Supabase, чтобы починить ошибку “new row violates row-level security policy for table 'user_progress'” и дать доступ к Storage.

1) Таблица user_progress

-- включить RLS
alter table public.user_progress enable row level security;

-- политика чтения и изменения только своих записей
create policy "user_progress_select_own"
on public.user_progress for select
using (user_id = auth.uid());

create policy "user_progress_insert_own"
on public.user_progress for insert
with check (user_id = auth.uid());

create policy "user_progress_update_own"
on public.user_progress for update
using (user_id = auth.uid())
with check (user_id = auth.uid());

create policy "user_progress_delete_own"
on public.user_progress for delete
using (user_id = auth.uid());

2) Buckets в Storage (например covers, assignment-images, help-images)

-- Разрешить всем читать
create policy "public read" on storage.objects for select
using (bucket_id in ('covers','assignment-images','help-images'));

-- Разрешить авторизованным загружать/удалять свои объекты
create policy "authenticated write" on storage.objects for insert
with check (bucket_id in ('covers','assignment-images','help-images') and auth.role() = 'authenticated');

create policy "authenticated update" on storage.objects for update
using (bucket_id in ('covers','assignment-images','help-images') and auth.role() = 'authenticated')
with check (bucket_id in ('covers','assignment-images','help-images') and auth.role() = 'authenticated');

create policy "authenticated delete" on storage.objects for delete
using (bucket_id in ('covers','assignment-images','help-images') and auth.role() = 'authenticated');

После применения этих политик пользователь сможет создавать/обновлять свой прогресс, а админ — загружать картинки (при этом всем можно их просматривать).
